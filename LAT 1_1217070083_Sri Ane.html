import pandas as pd
import string
from nltk.stem import PorterStemmer
from nltk.tokenize import word_tokenize
def getTokenizedList(input_df):
tokenizedList = []
# Initialize the stemmer
mystemmer = PorterStemmer()

for i in range(0,len(input_df)):
curText = input_df['Text'].iloc[i]
curText = curText.replace('\n', '')
curTokenized = word_tokenize(curText)
tokenizedList.append(curTokenized)

# Remove punctuation
tokenizedListWithoutPunct = []
punctList = list(string.punctuation)

for curList in tokenizedList:
newList = [word.lower() for word in curList if word not in punctList]
tokenizedListWithoutPunct.append(newList)

# Stemming
tokenizedStemmed = []
for curList in tokenizedListWithoutPunct:
newList = [mystemmer.stem(word) for word in curList]
tokenizedStemmed.append(newList)

return tokenizedStemmed

def transformSentence(sent):

mystemmer = PorterStemmer()

sentTokenized = word_tokenize(sent)
# Remove punctuation and convert to lowercase
punctList = set(string.punctuation)
newSentList = [word.lower() for word in sentTokenized if word not in punctList]
# Stemming
tokenizedStemmed = [mystemmer.stem(word) for word in newSentList]
